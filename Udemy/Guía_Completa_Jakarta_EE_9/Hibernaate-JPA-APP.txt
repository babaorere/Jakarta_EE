.- Crear un paquete "com.app.entity" y dentro de este paquete, una clase vacia llamada "Cliente": 

.- Crear una carpeta y archivo: src/main/resources/META-INF/persistence.xml

<?xml version="1.0" encoding="utf-8" ?>
<persistence xmlns="http://xmlns.jcp.org/xml/ns/persistence" version="3.0">
    <persistence-unit name="ejemploJPA">
        <provider>org.hibernate.jpa.HibernatePersistenceProvider</provider>
        <class>com.app.entity.Cliente</class>
        <properties>
            <property name="jakarta.persistence.jdbc.driver" value="com.mysql.cj.jdbc.Driver" />
            <property name="jakarta.persistence.jdbc.url" value="jdbc:mysql://localhost:3306/jakarta_ee_9?serverTimezone=America/Santiago"/>
            <property name="jakarta.persistence.jdbc.user" value="root" />
            <property name="jakarta.persistence.jdbc.password" value="root" />
        </properties>
    </persistence-unit>
</persistence>
